# 如何使用
直接配置config中的内容即可
```
    var config = {
        "center": {"j": "116.276079", "w": "40.16096"}, //中心点的经纬度
        "attractions": [ //
            {"label":"明十三陵总神道","j":"116.232006","w":"40.263881"}, //每个景点的经纬度，名称
            {"label":"古长城","j":"115.984126","w":"40.344634"} ////每个景点的经纬度，名称
        ]
    };
```
